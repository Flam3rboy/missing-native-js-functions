Object

forEach

callback is called for every element in object

forEach(callback: (element: any, index?: string) => any): void;

map

callback is called for every element in object and the result is returned as a new object

map(callback: (element: any, index?: string) => this): void;

equals

checks if this Objects is the same with the other, WARNING this won't work with circular objects

equals(other: any): boolean;

keys

returns keys of object itself

keys(): string[];

values

returns values of object itself

values(): any[];

entries

returns a nested array of key and corresponding value of object itself

entries(): Array<[string, any]>;

merge

returns a new object deeply merged with obj, the current will overwrite obj, if obj has the same property. Notice will not merge classes

merge(obj: any): any;

stringify

parses the object to an string (trows errors if invalid input)

stringify(): string;
